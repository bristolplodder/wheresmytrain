  def get_connection()
    conn_hdrs = {"accept-version" => "1.1",    # 1.1 only
      "host" => virt_host,                     # the vhost
    }
    conn_hash = { :hosts => [
      {:login => login, :passcode => passcode, :host => host, :port => port},
      ],
      :connect_headers => conn_hdrs,
    }
    conn = Stomp::Connection.new(conn_hash)
  end
